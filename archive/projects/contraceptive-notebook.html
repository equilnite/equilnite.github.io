Data Set Information:¶

This dataset is a subset of the 1987 National Indonesia Contraceptive Prevalence Survey. The samples are married women who were either not pregnant or do not know if they were at the time of interview. The problem is to predict the current contraceptive method choice (no use, long-term methods, or short-term methods) of a woman based on her demographic and socio-economic characteristics.

Attribute Information:¶

  1. Wife's age (numerical)
  2. Wife's education (categorical) 1=low, 2, 3, 4=high
  3. Husband's education (categorical) 1=low, 2, 3, 4=high
  4. Number of children ever born (numerical)
  5. Wife's religion (binary) 0=Non-Islam, 1=Islam
  6. Wife's now working? (binary) 0=Yes, 1=No
  7. Husband's occupation (categorical) 1, 2, 3, 4
  8. Standard-of-living index (categorical) 1=low, 2, 3, 4=high
  9. Media exposure (binary) 0=Good, 1=Not good
  10. Contraceptive method used (class attribute) 1=No-use, 2=Long-term, 3=Short-term

import sklearn
import sklearn.model_selection as ms
import sklearn.linear_model
from sklearn.linear_model import LogisticRegression
from sklearn import metrics

# current categorical features being used in our models
features = ['wife_religion', 'wife_work', 'husband_occupation', 'husband_education', 'standard_living', 'media_exposure', 'wife_education']
# current categorical features being excluded in our models
exclude = []

def splitDummies(df, exclude):
    """
    Split a data frame with all the features in our given data into the response and explanatory variables, 
    and one-hot encode every categorical variable in the response.
    This also excludes features from the resulting X df that aren in the global EXCLUDE list
    """
    X = pd.get_dummies(df.drop(["contraceptive"] + exclude, axis=1), columns=features)
    y = df.contraceptive
    return X, y

df = pd.read_csv("data.csv")

# split our data into train and test sets
train, test = ms.train_test_split(df, test_size=0.2, random_state=1)

# further split our sets into X and y sets with dummy variables
X_train, y_train = splitDummies(train, exclude)
X_test, y_test = splitDummies(test, exclude)

# Logistic Regression

print('Logistic Regression')
lr_model = LogisticRegression(max_iter=4000, solver="lbfgs")

lr_model.fit(X_train, y_train)


lr_train_accuracy = np.mean(lr_model.predict(X_train) == y_train)
lr_train_cross_accuracy = np.mean(ms.cross_val_score(lr_model, X_train, y_train, cv=5))
lr_test_accuracy = np.mean(lr_model.predict(X_test) == y_test)
print("Train accuracy:", round(lr_train_accuracy * 100, 2))
print("Cross validation accuracy:", round(lr_train_cross_accuracy * 100, 2))
print("Test accuracy:", round(lr_test_accuracy * 100, 2))

# Decision tree

print('\nDecision Tree')
from sklearn import tree
dt_model = tree.DecisionTreeClassifier(max_depth = 6)
dt_model.fit(X_train, y_train)

dt_train_accuracy = np.mean(dt_model.predict(X_train) == y_train)
dt_train_cross_accuracy = np.mean(ms.cross_val_score(dt_model, X_train, y_train, cv=5))
dt_test_accuracy = np.mean(dt_model.predict(X_test) == y_test)
print("Train accuracy:", round(dt_train_accuracy * 100, 2))
print("Cross validation accuracy:", round(dt_train_cross_accuracy * 100, 2))
print("Test accuracy:", round(dt_test_accuracy * 100, 2))

# Naive model
print('\nNaive')
print("Test accuracy:", round(sum(y_test == 1) / len(y_test) * 100, 2))
